## List Transactions

<Endpoint method="GET" path="/v1/transactions" />

Retrieve a paginated list of transactions with optional filtering by account or block height.

<Info>
  This endpoint supports pagination and filtering. Use the `limit` and `offset` parameters to navigate through results.
</Info>

### Query Parameters

<ParamField query="limit" type="integer" default="50">
  Number of transactions to return per request. Maximum: 100.
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of transactions to skip for pagination.
</ParamField>

<ParamField query="account" type="string">
  Filter transactions by account ID (e.g., `example.near`). Returns transactions where the account is either sender or receiver.
</ParamField>

<ParamField query="block_height" type="integer">
  Filter transactions by specific block height.
</ParamField>

    type TransactionClient struct {

        Client  *http.Client
    }

    func NewTransactionClient(apiKey string) *TransactionClient {
        return &TransactionClient{
            BaseURL: "https://api.shadowcorp.dev",
            APIKey:  apiKey,

    func (c *TransactionClient) GetTransactions(limit, offset int) (map[string]interface{}, error) {
        params := url.Values{}
        params.Add("limit", fmt.Sprintf("%d", limit))
        params.Add("offset", fmt.Sprintf("%d", offset))
        
        return c.makeRequest("/v1/transactions", params)
    }

    func (c *TransactionClient) GetAccountTransactions(accountID string) (map[string]interface{}, error) {
        params := url.Values{}
        params.Add("account", accountID)
        
        return c.makeRequest("/v1/transactions", params)
    }

    func (c *TransactionClient) makeRequest(path string, params url.Values) (map[string]interface{}, error) {
        reqURL := fmt.Sprintf("%s%s?%s", c.BaseURL, path, params.Encode())
        
        req, err := http.NewRequest("GET", reqURL, nil)
        if err != nil {
            return nil, err
        }
        
        req.Header.Add("X-API-Key", c.APIKey)
        
        resp, err := c.Client.Do(req)
        if err != nil {
            return nil, err
        }
        defer resp.Body.Close()

        var result map[string]interface{}
        json.NewDecoder(resp.Body).Decode(&result)
        return result, nil
    }

    func main() {

### Response Format

<ResponseField name="data" type="array" required>
  Array of transaction objects.
  
  <Expandable title="Transaction Object">
    <ResponseField name="tx_hash" type="string">
      Unique transaction hash identifier (base58 encoded).
    </ResponseField>
    
    <ResponseField name="block_timestamp" type="string">
      ISO 8601 timestamp when the transaction was included in a block.
    </ResponseField>
    
    <ResponseField name="block_height" type="integer">
      The block height where this transaction was included.
    </ResponseField>
    
    <ResponseField name="signer_account_id" type="string">
      Account ID that signed and sent the transaction.
    </ResponseField>
    
    <ResponseField name="receiver_account_id" type="string">
      Account ID that received or executed the transaction.
    </ResponseField>
    
    <ResponseField name="status" type="string">
      Transaction status. Possible values: `success`, `pending`, `failed`.
    </ResponseField>
    
    <ResponseField name="description" type="string">
      Human-readable description of the transaction action.
    </ResponseField>
    
    <ResponseField name="actions" type="array">
      Array of actions performed in this transaction (e.g., FunctionCall, Transfer, CreateAccount).
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total" type="integer" required>
  Total number of transactions matching the query (useful for pagination).
</ResponseField>

<ResponseField name="limit" type="integer" required>
  Number of transactions returned in this response.
</ResponseField>

<ResponseField name="offset" type="integer" required>
  Current pagination offset.
</ResponseField>

## Understanding Transaction Fields

<AccordionGroup>
  <Accordion title="Transaction Hash" icon="fingerprint">
    The `tx_hash` is a unique identifier for each transaction on the NEAR blockchain. It's a base58-encoded string that can be used to look up the transaction on block explorers or via the API.
    
    **Use case:** Store this hash to reference and track specific transactions over time.
  </Accordion>
  
  <Accordion title="Transaction Status" icon="traffic-light">
    The `status` field indicates the current state of the transaction:
    
    - **`success`**: Transaction was executed successfully and included in the blockchain
    - **`pending`**: Transaction is being processed (rare, usually only seen in real-time queries)
    - **`failed`**: Transaction failed during execution (contract error, insufficient gas, etc.)
    
    <Tip>
      Always check the status before processing transaction results in your application.
    </Tip>
  </Accordion>
  
  <Accordion title="Actions Array" icon="list-check">
    NEAR transactions can contain multiple actions. Common action types include:
    
    - **FunctionCall**: Calling a smart contract method
    - **Transfer**: Sending NEAR tokens
    - **CreateAccount**: Creating a new account
    - **DeployContract**: Deploying smart contract code
    - **DeleteAccount**: Deleting an account
    - **AddKey**: Adding an access key
    - **DeleteKey**: Removing an access key
    - **Stake**: Staking NEAR tokens with a validator
    
    Each action may have different fields depending on its type.
  </Accordion>
  
  <Accordion title="Pagination Strategy" icon="arrows-left-right">
    With millions of transactions, proper pagination is crucial:
    
    ```javascript
    async function getAllAccountTransactions(accountId) {
      const allTransactions = [];
      let offset = 0;
      const limit = 100; // Max limit for best performance
      
      while (true) {
        const response = await fetch(
          `https://api.shadowcorp.dev/v1/transactions?account=${accountId}&limit=${limit}&offset=${offset}`,
          { headers: { 'X-API-Key': 'YOUR_KEY' } }
        );
        const data = await response.json();
        
        allTransactions.push(...data.data);
        
        // Check if we've retrieved all transactions
        if (offset + limit >= data.total) {
          break;
        }
        
        offset += limit;
      }
      
      return allTransactions;
    }
    ```
  </Accordion>
</AccordionGroup>

## Batch Lookup Transactions

<Endpoint method="POST" path="/v1/transactions" />

Look up multiple transactions by their hashes in a single request. Maximum 20 hashes per request.

### Request Body

<ParamField body="tx_hashes" type="string[]" required>
  Array of transaction hashes to look up. Maximum 20 hashes per request.
</ParamField>

### Making Requests

<Tabs>
  <Tab title="cURL">
    ```bash
    curl -X POST https://api.shadowcorp.dev/v1/transactions \
      -H "X-API-Key: YOUR_KEY" \
      -H "Content-Type: application/json" \
      -d '{
        "tx_hashes": [
          "Dk6L1UbC16cWtBpjRPPRxBhF1wdVLU2CjFUabWeXtuud",
          "GGLBFP2XrrzvaJRjQaDGrZm54DNDeJCV7kVqPpnQHjpV"
        ]
      }'
    ```
  </Tab>
  
  <Tab title="JavaScript">

    func (c *TransactionClient) BatchLookupTransactions(hashes []string) ([]map[string]interface{}, error) {
        if len(hashes) > 20 {
            return nil, fmt.Errorf("maximum 20 transaction hashes per request")
        }
        
        payload := map[string]interface{}{
            "tx_hashes": hashes,
        }
        
        jsonData, err := json.Marshal(payload)
        if err != nil {
            return nil, err
        }
        
        req, err := http.NewRequest("POST", c.BaseURL+"/v1/transactions", bytes.NewBuffer(jsonData))
        if err != nil {
            return nil, err
        }
        
        req.Header.Add("X-API-Key", c.APIKey)
        req.Header.Add("Content-Type", "application/json")
        
        resp, err := c.Client.Do(req)
        if err != nil {
            return nil, err
        }
        defer resp.Body.Close()

        var result []map[string]interface{}
        json.NewDecoder(resp.Body).Decode(&result)
        return result, nil
    }

## Advanced Example: Transaction Monitor

Here's a complete example of a transaction monitoring system:

```javascript
class TransactionMonitor {
  constructor(apiKey) {
    this.apiKey = apiKey;
    this.baseUrl = 'https://api.shadowcorp.dev';
    this.lastCheckedHeight = null;
  }

  async monitorAccount(accountId, callback) {
    console.log(`ðŸ“¡ Starting monitor for account: ${accountId}`);
    
    // Initial fetch to get current state
    const initial = await this.getAccountTransactions(accountId, 1);
    this.lastCheckedHeight = initial.data[0]?.block_height || 0;
    
    // Poll every 10 seconds
    setInterval(async () => {
      await this.checkNewTransactions(accountId, callback);
    }, 10000);
  }

  async checkNewTransactions(accountId, callback) {
    try {
      const response = await this.getAccountTransactions(accountId, 50);
      
      // Filter for new transactions
      const newTxs = response.data.filter(
        tx => tx.block_height > this.lastCheckedHeight
      );
      
      if (newTxs.length > 0) {
        console.log(`ðŸ†• Found ${newTxs.length} new transaction(s)`);
        
        // Update last checked height
        this.lastCheckedHeight = Math.max(
          ...newTxs.map(tx => tx.block_height)
        );
        
        // Call callback for each new transaction
        newTxs.forEach(callback);
      }
    } catch (error) {
      console.error('Error checking transactions:', error);
    }
  }

  async getAccountTransactions(accountId, limit) {
    const response = await fetch(
      `${this.baseUrl}/v1/transactions?account=${accountId}&limit=${limit}`,
      {
        headers: { 'X-API-Key': this.apiKey }
      }
    );
    return await response.json();
  }
}

// Usage
const monitor = new TransactionMonitor('YOUR_KEY');

monitor.monitorAccount('example.near', (tx) => {
  console.log('New transaction detected:');
  console.log(`  Hash: ${tx.tx_hash}`);
  console.log(`  From: ${tx.signer_account_id}`);
  console.log(`  To: ${tx.receiver_account_id}`);
  console.log(`  Status: ${tx.status}`);
  console.log(`  Description: ${tx.description}`);
});
```
